\subsection{Oppkobling}
\begin{todolist}%[noitemsep]
    \item{Finn et egnet sted og sett opp en sikkerhetssone med ca. 3m radius fra spoleriggen}
    \item{Plasser spoleriggen på egnet sted}
    \iftoggle{INTERN}{
        % Intern
        \item{Koble ønsket hanske til spoleriggen}
    }{
        % Ekstern
    }
    
    \item{Plasser driveren 3m unna spoleriggen i egnet retning, innennfor synsrekkevidde til spoleriggen}
    \item{Plasser \signalkilde{ }og \pulsformer{ }på et egnet sted innen synsrekkevidde til driveren, spoleriggen, og publikum}
    \item{Plasser nødstoppen på egnet sted. Innenfor synsrekkevidde til spoleriggen, \signalkilde{ }og \pulsformer}
    \item{Koble 3,5mm jack -> phono mellom \signalkilde{ }og \pulsformer}
    \item{Koble optisk kabel fra den midterste utgangen til \pulsformer{ }og til driveren}
    \item{Koble kabelen til spoleriggen, men ikke i driveren}
    \item{Koble strøm til \signalkilde{ }og \pulsformer}
    \item{Koble driveren til nødstoppen (strømkabel)}
    \item{Sørg for at nødstoppbryteren er av (ikke strøm ut)}
    \item{Koble nødstoppen til egnet strømmuttak (min. 10A)}
    \item{Sjekk alle koblinger på nytt}
    \item{Gå igjennom sjekkliste for spolerigg}
    \item{Gå igjennom sjekkliste for \signalkilde/\pulsformer}
    \item{Gå igjennom sjekkliste for driveren}
\end{todolist}
